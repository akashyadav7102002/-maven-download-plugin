3   import java.net.URI;

5   /**
6    * Common utilities used by the {@link WGetMojo}
7    */
8   public final class FileNameUtils {
9   
10      /**
11       * Attempts to construct the target file name based on an URI as the relative resource name
12       * or, if the root resource is requested, the host name extracted from the URI.
13       * @param uri uri to extract the output name from
14       * @return output file name based on the URI
15       */
16      public static String getOutputFileName(URI uri) {
17          return uri.getPath().isEmpty() || uri.getPath().equals("/")
18                  ? uri.getHost()
19                  : uri.getPath().substring(uri.getPath().lastIndexOf('/') + 1);
20      }
