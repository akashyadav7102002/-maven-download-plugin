277     /**
278      * Method call when the mojo is executed for the first time.
279      *
280      * @throws MojoExecutionException if an error is occuring in this mojo.
281      * @throws MojoFailureException   if an error is occuring in this mojo.
282      */
283     @Override
284     public void execute() throws MojoExecutionException, MojoFailureException {
285         if (this.skip) {
286             getLog().info("maven-download-plugin:wget skipped");
287             return;
288         }
289 
290         if (this.runOnlyAtRoot && !this.session.getCurrentProject().isExecutionRoot()) {
291             getLog().info("maven-download-plugin:wget skipped (not project root)");
292             return;
293         }
294 
295         if (isNotBlank(this.serverId) && (isNotBlank(this.username) || isNotBlank(this.password))) {
296             throw new MojoExecutionException("Specify either serverId or username/password, not both");
297         }
298 
299         if (this.session.getSettings() == null) {
300             getLog().warn("settings is null");
301         }
302         if (this.session.getSettings().isOffline()) {
303             getLog().debug("maven-download-plugin:wget offline mode");
304         }
305         getLog().debug("Got settings");
306         if (this.retries < 1) {
307             throw new MojoFailureException("retries must be at least 1");
308         }
309 
310         // PREPARE
311         if (this.outputFileName == null) {
312             try {
313                 this.outputFileName = new File(this.uri.toURL().getPath()).getName();
314             } catch (Exception ex) {
315                 throw new MojoExecutionException("Invalid URL", ex);
316             }
317         }
318         if (!this.skipCache) {
319             if (this.cacheDirectory == null) {
320                 this.cacheDirectory = new File(this.session.getLocalRepository()
321                         .getBasedir(), ".cache/download-maven-plugin");
322             } else if (this.cacheDirectory.exists() && !this.cacheDirectory.isDirectory()) {
323                 throw new MojoFailureException(String.format("cacheDirectory is not a directory: "
324                         + this.cacheDirectory.getAbsolutePath()));
325             }
326             getLog().debug("Cache is: " + this.cacheDirectory.getAbsolutePath());
327 
328         } else {
329             getLog().debug("Cache is skipped");
330         }
331         this.outputDirectory.mkdirs();
332         final File outputFile = new File(this.outputDirectory, this.outputFileName);
333         final Lock fileLock = FILE_LOCKS.computeIfAbsent(
334             outputFile.getAbsolutePath(), ignored -> new ReentrantLock()
335         );
336 
337         final Checksums checksums = new Checksums(
338             this.md5, this.sha1, this.sha256, this.sha512, this.getLog()
339         );
340         // DO
341         boolean lockAcquired = false;
342         try {
343             lockAcquired = fileLock.tryLock(
344                 this.maxLockWaitTime, TimeUnit.MILLISECONDS
345             );
346             if (!lockAcquired) {
347                 final String message = String.format(
348                     "Could not acquire lock for File: %s in %dms",
349                     outputFile, this.maxLockWaitTime
350                 );
351                 if (this.failOnError) {
352                     throw new MojoExecutionException(message);
353                 } else {
354                     getLog().warn(message);
355                     return;
356                 }
357             }
358             boolean haveFile = outputFile.exists();
359             if (haveFile) {
360                 boolean checksumMatch = true;
361                 if (this.alwaysVerifyChecksum || this.checkSignature) {
362                     try {
363                         checksums.validate(outputFile);
364                     } catch (final MojoFailureException e) {
365                         getLog().warn("The local version of file " + outputFile.getName()
366                                 + " doesn't match the expected checksum. "
367                                 + "You should consider checking the specified checksum is correctly set.");
368                         checksumMatch = false;
369                     }
370                 }
371                 if (!checksumMatch || this.overwrite) {
372                     outputFile.delete();
373                     haveFile = false;
374                 } else {
375                     getLog().info("File already exist, skipping");
376                 }
377             }
378 
379             if (!haveFile) {
380                 if (this.session.getRepositorySession().isOffline()) {
381                     if (this.failOnError) {
382                         throw new MojoExecutionException("No file in cache and maven is in offline mode");
383                     } else {
384                         getLog().warn("Ignoring download failure.");
385                     }
386                 }
387                 boolean done = false;
388                 for (int retriesLeft = this.retries; retriesLeft > 0; --retriesLeft) {
389                     try {
390                         this.doGet(outputFile);
391                         checksums.validate(outputFile);
392                         done = true;
393                     } catch (DownloadFailureException ex) {
394                         // treating HTTP codes >= 500 as transient and thus always retriable
395                         if (this.failOnError && ex.getHttpCode() < 500) {
396                             throw new MojoExecutionException(ex.getMessage(), ex);
397                         } else {
398                             getLog().warn(ex.getMessage());
399                         }
400                     } catch (IOException ex) {
401                         if (this.failOnError) {
402                             throw new MojoExecutionException(ex.getMessage(), ex);
403                         } else {
404                             getLog().warn(ex.getMessage());
405                         }
406                     }
407                     if (!done) {
408                         getLog().warn("Retrying (" + (retriesLeft - 1) + " more)");
409                     }
410                 }
411                 if (!done) {
412                     if (this.failOnError) {
413                         throw new MojoFailureException("Could not get content after " + this.retries + " failed attempts.");
414                     } else {
415                         getLog().warn("Ignoring download failure(s).");
416                         return;
417                     }
418                 }
419             }
420             if (this.unpack) {
421                 unpack(outputFile);
422                 this.buildContext.refresh(this.outputDirectory);
423             } else {
424             	this.buildContext.refresh(outputFile);
425             }
426         } catch (MojoExecutionException e) {
427             throw e;
428         } catch (IOException ex) {
429             throw new MojoExecutionException("IO Error: ", ex);
430         } catch (NoSuchArchiverException e) {
431             throw new MojoExecutionException("No such archiver: " + e.getMessage());
432         } catch (Exception e) {
433             throw new MojoExecutionException("General error: ", e);
434         } finally {
435             if (lockAcquired) {
436                 fileLock.unlock();
437             }
438         }
439     }

442     private void unpack(File outputFile) throws NoSuchArchiverException {
443         UnArchiver unarchiver = this.archiverManager.getUnArchiver(outputFile);
444         unarchiver.setSourceFile(outputFile);
445         if (isFileUnArchiver(unarchiver)) {
446             unarchiver.setDestFile(new File(this.outputDirectory, this.outputFileName.substring(0,
447                     this.outputFileName.lastIndexOf('.'))));
448         } else {
449             unarchiver.setDestDirectory(this.outputDirectory);
450         }
451         unarchiver.setFileMappers(this.fileMappers);
452         unarchiver.extract();
453         outputFile.delete();
454     }
455 
456     private boolean isFileUnArchiver(final UnArchiver unarchiver) {
457         return unarchiver instanceof  BZip2UnArchiver ||
458                 unarchiver instanceof GZipUnArchiver ||
459                 unarchiver instanceof SnappyUnArchiver ||
460                 unarchiver instanceof XZUnArchiver;
461     }

463     private static RemoteRepository createRemoteRepository(String serverId, URI uri)
464     {
465         return new RemoteRepository.Builder(isBlank(serverId)
466                     ? null
467                     : serverId,
468                 isBlank(serverId)
469                     ? uri.getScheme()
470                     : null,
471                 isBlank(serverId)
472                     ? uri.getHost()
473                     : null)
474                 .build();
475     }

477     private void doGet(final File outputFile) throws IOException, MojoExecutionException {
478         final HttpFileRequester.Builder fileRequesterBuilder = new HttpFileRequester.Builder();
479 
480         final RemoteRepository repository = createRemoteRepository(this.serverId, this.uri);
481 
482         // set proxy if present
483         Optional.ofNullable(this.session.getRepositorySession().getProxySelector())
484                 .map(selector -> selector.getProxy(repository))
485                 .ifPresent(proxy -> addProxy(fileRequesterBuilder, repository, proxy));
486 
487         Optional.ofNullable(this.session.getRepositorySession().getAuthenticationSelector())
488                 .map(selector -> selector.getAuthentication(repository))
489                 .ifPresent(auth -> addAuthentication(fileRequesterBuilder, repository, auth));
490 
491         if (!this.skipCache) {
492             fileRequesterBuilder.withCacheDir(this.cacheDirectory);
493         }
494 
495         final HttpFileRequester fileRequester = fileRequesterBuilder
496                 .withProgressReport(this.session.getSettings().isInteractiveMode()
497                         ? new LoggingProgressReport(this.getLog())
498                         : new SilentProgressReport(this.getLog()))
499                 .withConnectTimeout(this.readTimeOut)
500                 .withSocketTimeout(this.readTimeOut)
501                 .withUri(this.uri)
502                 .withUsername(this.username)
503                 .withPassword(this.password)
504                 .withServerId(this.serverId)
505                 .withPreemptiveAuth(this.preemptiveAuth)
506                 .withMavenSession(this.session)
507                 .withRedirectsEnabled(this.followRedirects)
508                 .withLog(this.getLog())
509                 .build();
510         fileRequester.download(outputFile, getAdditionalHeaders());
511     }

513     private void addProxy(final HttpFileRequester.Builder fileRequesterBuilder,
514                           final RemoteRepository repository,
515                           final Proxy proxy) {
516         fileRequesterBuilder.withProxyHost(proxy.getHost());
517         fileRequesterBuilder.withProxyPort(proxy.getPort());
518 
519         final RemoteRepository proxyRepo = new RemoteRepository.Builder(repository)
520                 .setProxy(proxy)
521                 .build();
522 
523         try ( final AuthenticationContext ctx = AuthenticationContext.forProxy(this.session.getRepositorySession(),
524                 proxyRepo) ) {
525             fileRequesterBuilder.withProxyUserName(ctx.get(AuthenticationContext.USERNAME));
526             fileRequesterBuilder.withProxyPassword(ctx.get(AuthenticationContext.PASSWORD));
527             fileRequesterBuilder.withNtlmDomain(ctx.get(AuthenticationContext.NTLM_DOMAIN));
528             fileRequesterBuilder.withNtlmHost(ctx.get(AuthenticationContext.NTLM_WORKSTATION));
529         }
530     }
531 
532     private void addAuthentication(final HttpFileRequester.Builder fileRequesterBuilder,
533                                    final RemoteRepository repository,
534                                    final Authentication authentication) {
535         final RemoteRepository authRepo = new RemoteRepository.Builder(repository)
536                 .setAuthentication(authentication)
537                 .build();
538         try (final AuthenticationContext authCtx = AuthenticationContext.forRepository(
539                 this.session.getRepositorySession(),
540                 authRepo)) {
541             final String username = authCtx.get(AuthenticationContext.USERNAME);
542             final String password = authCtx.get(AuthenticationContext.PASSWORD);
543             final String ntlmDomain = authCtx.get(AuthenticationContext.NTLM_DOMAIN);
544             final String ntlmHost = authCtx.get(AuthenticationContext.NTLM_WORKSTATION);
545 
546             getLog().debug("providing custom authentication");
547             getLog().debug("username: " + username + " and password: ***");
548 
549             fileRequesterBuilder.withUsername(username);
550             fileRequesterBuilder.withPassword(password);
551             fileRequesterBuilder.withNtlmDomain(ntlmDomain);
552             fileRequesterBuilder.withNtlmHost(ntlmHost);
553         }
554     }

556     private List<Header> getAdditionalHeaders() {
557         return headers.entrySet().stream()
558                 .map(pair -> new BasicHeader(pair.getKey(), pair.getValue()))
559                 .collect(Collectors.toList());
560     }
